78  nmeaPARSER *parser,
79  const char *buff, int buff_sz,
80  nmeaINFO *info
81  )
82 {

103  nmea_GPGSV2info((nmeaGPGSV *)pack, info);
104  break;

106  nmea_GPRMC2info((nmeaGPRMC *)pack, info);
107  break;
108  case GPVTG:
109  nmea_GPVTG2info((nmeaGPVTG *)pack, info);
110  break;
111  };
112 
113  free(pack);
114  }
115

123 int nmea_parser_real_push(nmeaPARSER *parser, const char *buff, int buff_sz)
124 {
125  int nparsed = 0, crc, sen_sz, ptype;
126  nmeaParserNODE *node = 0;
127 
128  NMEA_ASSERT(parser && parser->buffer);
129 
130  /* clear unuse buffer (for debug) */
131  /*
132  memset(
133  parser->buffer + parser->buff_use, 0,
134  parser->buff_size - parser->buff_use
135  );
136  */
137 
138  /* add */
139  if(parser->buff_use + buff_sz >= parser->buff_size)
140  nmea_parser_buff_clear(parser);
141 
142  memcpy(parser->buffer + parser->buff_use, buff, buff_sz);
143  parser->buff_use += buff_sz;
144 
145  /* parse */
146  for(;;node = 0)
147  {
148  sen_sz = nmea_find_tail(
149  (const char *)parser->buffer + nparsed,
150  (int)parser->buff_use - nparsed, &crc);
151 
152  if(!sen_sz)
153  {
154  if(nparsed)
155  memcpy(
156  parser->buffer,
157  parser->buffer + nparsed,
158  parser->buff_use -= nparsed);
159  break;
160  }
161  else if(crc >= 0)
162  {
163  ptype = nmea_pack_type(
164  (const char *)parser->buffer + nparsed + 1,
165  parser->buff_use - nparsed - 1);
166 
167  if(0 == (node = malloc(sizeof(nmeaParserNODE))))
168  goto mem_fail;
169 
170  node->pack = 0;
171 
172  switch(ptype)
173  {
174  case GPGGA:
175  if(0 == (node->pack = malloc(sizeof(nmeaGPGGA))))
176  goto mem_fail;
177  node->packType = GPGGA;
178  if(!nmea_parse_GPGGA(
179  (const char *)parser->buffer + nparsed,
180  sen_sz, (nmeaGPGGA *)node->pack))
181  {
182  free(node);
183  node = 0;
184  }
185  break;
186  case GPGSA:
187  if(0 == (node->pack = malloc(sizeof(nmeaGPGSA))))
188  goto mem_fail;
189  node->packType = GPGSA;
190  if(!nmea_parse_GPGSA(
191  (const char *)parser->buffer + nparsed,
192  sen_sz, (nmeaGPGSA *)node->pack))
193  {
194  free(node);
195  node = 0;
196  }
197  break;
198  case GPGSV:
199  if(0 == (node->pack = malloc(sizeof(nmeaGPGSV))))
200  goto mem_fail;
201  node->packType = GPGSV;
202  if(!nmea_parse_GPGSV(
203  (const char *)parser->buffer + nparsed,
204  sen_sz, (nmeaGPGSV *)node->pack))
205  {
206  free(node);
207  node = 0;
208  }
209  break;
210  case GPRMC:
211  if(0 == (node->pack = malloc(sizeof(nmeaGPRMC))))
212  goto mem_fail;
213  node->packType = GPRMC;
214  if(!nmea_parse_GPRMC(
215  (const char *)parser->buffer + nparsed,
216  sen_sz, (nmeaGPRMC *)node->pack))
217  {
218  free(node);
219  node = 0;
220  }
221  break;
222  case GPVTG:
223  if(0 == (node->pack = malloc(sizeof(nmeaGPVTG))))
224  goto mem_fail;
225  node->packType = GPVTG;
226  if(!nmea_parse_GPVTG(
227  (const char *)parser->buffer + nparsed,
228  sen_sz, (nmeaGPVTG *)node->pack))
229  {
230  free(node);
231  node = 0;
232  }
233  break;
234  default:
235  free(node);
236  node = 0;
237  break;
238  };
239 
240  if(node)
241  {
242  if(parser->end_node)
243  ((nmeaParserNODE *)parser->end_node)->next_node = node;
244  parser->end_node = node;
245  if(!parser->top_node)
246  parser->top_node = node;
247  node->next_node = 0;
248  }
249  }
250 
251  nparsed += sen_sz;
252  }
253 
254  return nparsed;
255 
256 mem_fail:
257  if(node)
258  free(node);
259 
260  nmea_error("Insufficient memory!");
261 
262  return -1;
263 }

269 int nmea_parser_push(nmeaPARSER *parser, const char *buff, int buff_sz)
270 {
271  int nparse, nparsed = 0;
272 
273  do
274  {

276  nparse = parser->buff_size;
277  else
278  nparse = buff_sz;
279 
280  nparsed += nmea_parser_real_push(
281  parser, buff, nparse);
282 
283  buff_sz -= nparse;
284 
285  } while(buff_sz);
286 
287  return nparsed;
