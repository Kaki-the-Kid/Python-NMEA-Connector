44 static DWORD WINAPI VGD_TimerThread(PVOID pContext)
45 {
46  PVGD_INFO pHWHead = (PVGD_INFO)(pContext);
47 
48  DEBUGMSG(ZONE_INIT,
49  (TEXT("+VGD_TimerThread: pHWHead: 0x%X\r\n"), pHWHead));
50 
51  do {
52 
53  VGD_GenCycle(pContext);
54 
55  } while(WAIT_OBJECT_0 != WaitForSingleObject(
56  pHWHead->stop_thread_event, pHWHead->timer_timeout));
57 
58  SetEvent(pHWHead->kill_thread_event);
59 
60  DEBUGMSG(ZONE_INIT,
61  (TEXT("-VGD_TimerThread: pHWHead: 0x%X\r\n"), pHWHead));
62

67 BOOL VGD_StartTimerThread(PVOID pHead)
68 {
69  PVGD_INFO pHWHead = (PVGD_INFO)(pHead);
70 
71  DEBUGMSG(ZONE_INIT,
72  (TEXT("+VGD_StartTimerThread: pHWHead: 0x%X\r\n"), pHWHead));
73 
74  VGD_BEG_OP(pHWHead);
75 
76  if(!pHWHead->timer_thread)
77  {
78  pHWHead->stop_thread_event =
79  CreateEvent(0, FALSE, FALSE, NULL);
80  pHWHead->kill_thread_event =
81  CreateEvent(0, FALSE, FALSE, NULL);
82  pHWHead->timer_thread =
83  CreateThread(NULL, 0, VGD_TimerThread, pHead, 0, NULL);
84 
85  if(pHWHead->timer_thread)
86  CeSetThreadPriority(pHWHead->timer_thread, VGD_THREAD_PRIORITY);
87  else
88  {
89  CloseHandle(pHWHead->stop_thread_event);
90  CloseHandle(pHWHead->kill_thread_event);
91  pHWHead->stop_thread_event = NULL;
92  pHWHead->kill_thread_event = NULL;
93  }
94  }
95 
96  VGD_END_OP(pHWHead);
97 
98  DEBUGMSG(ZONE_INIT,
99  (TEXT("-VGD_StartTimerThread: pHWHead: 0x%X, timer_thread: 0x%X\r\n"), pHWHead->timer_thread));
100 
101  return NULL != pHWHead->timer_thread;
102 }

127  pHWHead->kill_thread_event = NULL;
128  pHWHead->stop_thread_event = NULL;
129  pHWHead->timer_thread = NULL;
130  }
131 
132  VGD_END_OP(pHWHead);
133 
134  DEBUGMSG(ZONE_INIT,
135  (TEXT("-VGD_StopTimerThread\r\n"), pHWHead));
136 }

385 ULONG SL_RxIntr(PVOID pHead, PUCHAR pRxBuffer, ULONG *pBufflen)
386 {
387 #if (VGD_ABOUT_ENABLE == 1)
388  static CHAR drv_about_string[] = VGD_ABOUT_STRING;
389  static UCHAR drv_about_len = sizeof(drv_about_string);
390  BOOL add_about = FALSE;
391 #endif
392 
393  PVGD_INFO pHWHead = (PVGD_INFO)pHead;
394  ULONG dropped = 0;
395 
396  if(!pBufflen || 0 == *pBufflen)
397  return (0);
398 
399  DEBUGMSG (ZONE_READ, (TEXT("+SL_RxIntr - 0x%X, 0x%X, %d.\r\n"),
400  pHead, pRxBuffer, *pBufflen));
401 
402 #if VGD_ABOUT_ENABLE
403  if((pHWHead->flags & VGD_FLG_ABOUTSTR) && *pBufflen > drv_about_len)
404  {
405  add_about = TRUE;
406  memcpy(pRxBuffer, &drv_about_string, drv_about_len);
407  *pBufflen -= drv_about_len;
408  pRxBuffer += drv_about_len;
409  pHWHead->flags &= ~VGD_FLG_ABOUTSTR;
410  }
411 #endif
412 
414  // read
415 
416  VGD_BEG_OP(pHWHead);
417 
418  *pBufflen = nmea_generate_from(
419  pRxBuffer, *pBufflen,
420  pHWHead->nmea_info, pHWHead->nmea_gen,
421  GPGGA | GPGSA | GPGSV | GPRMC | GPVTG
422  );
423 
424  VGD_END_OP(pHWHead);
425 
426 #if VGD_ABOUT_ENABLE
427  if(add_about)
428  *pBufflen += drv_about_len;
429 #endif
430 
431  pHWHead->curr_interruptIDs &= ~INTR_RX;
432 
433  DEBUGMSG (ZONE_READ, (TEXT("-SL_RxIntr - rx'ed %d.\r\n"), *pBufflen));
434 
435  return dropped;
436 }

612 //=========================================================
613 BOOL SL_Ioctl(PVOID pHead, DWORD dwCode,PBYTE pBufIn,DWORD dwLenIn,
614  PBYTE pBufOut,DWORD dwLenOut,PDWORD pdwActualOut)
615 {
616  BOOL RetVal = TRUE;
617  PVGD_INFO pHWHead = (PVGD_INFO)pHead;
618  int gen_index;
619 
620  DEBUGMSG (ZONE_FUNCTION, (TEXT("+SL_Ioctl 0x%X, 0x%X\r\n"), pHead, pHWHead));
621 
622  VGD_BEG_OP(pHead);
623 
624  switch (dwCode)
625  {
626  case VGD_IOCTL_SETLAT:
627  DEBUGMSG (ZONE_IOCTL,
628  (TEXT("SL_Ioctl (IOCTL): VGD_IOCTL_SETLAT...\r\n")));
629  if ( (dwLenIn != sizeof(double)) || (NULL == pBufIn) ) {
630  SetLastError (ERROR_INVALID_PARAMETER);
631  RetVal = FALSE;
632  DEBUGMSG (ZONE_IOCTL, (TEXT(" Invalid parameter\r\n")));
633  break;
634  }
635  pHWHead->nmea_info->lat = fabs(*(double *)pBufIn);
636  pHWHead->nmea_info->ns = nmea_ns_get(*(double *)pBufIn);
637  DEBUGMSG (ZONE_IOCTL,
638  (TEXT("SL_Ioctl (IOCTL): lat = %f\r\n"), pHWHead->nmea_info->lat));
639  break;
640 
641  case VGD_IOCTL_SETLON:
642  DEBUGMSG (ZONE_IOCTL,
643  (TEXT("SL_Ioctl (IOCTL): VGD_IOCTL_SETLON...\r\n")));
644  if ( (dwLenIn != sizeof(double)) || (NULL == pBufIn) ) {
645  SetLastError (ERROR_INVALID_PARAMETER);
646  RetVal = FALSE;
647  DEBUGMSG (ZONE_IOCTL, (TEXT(" Invalid parameter\r\n")));
648  break;
649  }
650  pHWHead->nmea_info->lon = fabs(*(double *)pBufIn);
651  pHWHead->nmea_info->ew = nmea_ew_get(*(double *)pBufIn);
652  DEBUGMSG (ZONE_IOCTL,
653  (TEXT("SL_Ioctl (IOCTL): lon = %f\r\n"), pHWHead->nmea_info->lon));
654  break;
655 
656  case VGD_IOCTL_GENIDX:
657  DEBUGMSG (ZONE_IOCTL,
658  (TEXT("SL_Ioctl (IOCTL): VGD_IOCTL_GENIDX...\r\n")));
659  if ( (dwLenIn < sizeof(int)) || (NULL == pBufIn) ) {
660  SetLastError (ERROR_INVALID_PARAMETER);
661  RetVal = FALSE;
662  DEBUGMSG (ZONE_IOCTL, (TEXT(" Invalid parameter\r\n")));
663  break;
664  }
665
